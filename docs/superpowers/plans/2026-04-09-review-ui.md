# Post-Mortem Review UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a "Review" tab to the existing demo UI so engineers can browse, inspect, and approve/reject/request-changes on generated post-mortems, with reviewer name persisted to `reviews.json`.

**Architecture:** Three new FastAPI routes read from `demo/mock_postmortems.json` and read/write `demo/reviews.json`. Effective status per post-mortem is derived at query time from the most recent review entry. A new Review tab in `demo/static/index.html` renders a filterable list (left) and a detail + action bar (right), all vanilla JS with no build step.

**Tech Stack:** Python 3.11, FastAPI, Pydantic v2, vanilla JS, pytest + starlette TestClient

**New helper functions** (add after `_calc_duration`):

```python
_ACTION_TO_STATUS = {
    "approve": "approved",
    "request_changes": "needs_changes",
    "reject": "rejected",
}


def _load_postmortems() -> list[dict]:
    return json.loads(MOCK_POSTMORTEMS_FILE.read_text())


def _load_reviews() -> list[dict]:
    if not REVIEWS_FILE.exists():
        return []
    return json.loads(REVIEWS_FILE.read_text())


def _derive_status(incident_id: str, reviews: list[dict]) -> str:
    matching = [r for r in reviews if r["incident_id"] == incident_id]
    if not matching:
        return "draft"
    latest = max(matching, key=lambda r: r["timestamp"])
    return _ACTION_TO_STATUS.get(latest["action"], latest["action"])
```

**New routes** (add at the end of `demo/app.py`):

```python
@app.get("/postmortems")
async def list_postmortems(status: str = "all", severity: str = "all") -> list[dict]:
    postmortems = _load_postmortems()
    reviews = _load_reviews()
    result = []
    for pm in postmortems:
        inc = pm["incident"]
        derived_status = _derive_status(inc["id"], reviews)
        if status != "all" and derived_status != status:
            continue
        if severity != "all" and inc["severity"] != severity:
            continue
        matching = sorted(
            [r for r in reviews if r["incident_id"] == inc["id"]],
            key=lambda r: r["timestamp"],
        )
        last_review = matching[-1] if matching else None
        result.append({
            "id": inc["id"],
            "title": inc["title"],
            "severity": inc["severity"],
            "started_at": inc["started_at"],
            "resolution_duration_minutes": pm["resolution_duration_minutes"],
            "evaluator_total": pm["evaluator_scores"]["total"],
            "status": derived_status,
            "last_reviewer": last_review["reviewer"] if last_review else None,
            "last_reviewed_at": last_review["timestamp"] if last_review else None,
        })
    result.sort(key=lambda x: x["started_at"], reverse=True)
    return result


@app.get("/postmortems/{incident_id}", response_model=None)
async def get_postmortem(incident_id: str) -> dict | JSONResponse:
    postmortems = _load_postmortems()
    reviews = _load_reviews()
    pm = next((p for p in postmortems if p["incident"]["id"] == incident_id), None)
    if pm is None:
        return JSONResponse({"error": f"Post-mortem '{incident_id}' not found"}, status_code=404)
    matching_reviews = sorted(
        [r for r in reviews if r["incident_id"] == incident_id],
        key=lambda r: r["timestamp"],
    )
    return {
        "postmortem": pm,
        "evaluator_scores": pm["evaluator_scores"],
        "review_history": matching_reviews,
        "status": _derive_status(incident_id, reviews),
    }


@app.post("/postmortems/{incident_id}/review", response_model=None)
async def review_postmortem(incident_id: str, body: ReviewRequest) -> dict | JSONResponse:
    if not body.reviewer.strip():
        raise HTTPException(status_code=422, detail="Reviewer name is required")
    if body.action == "request_changes" and not (body.comment and body.comment.strip()):
        raise HTTPException(status_code=422, detail="Comment is required when requesting changes")
    postmortems = _load_postmortems()
    if not any(p["incident"]["id"] == incident_id for p in postmortems):
        return JSONResponse({"error": f"Post-mortem '{incident_id}' not found"}, status_code=404)
    reviews = _load_reviews()
    now = datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")
    entry: dict = {
        "review_id": str(uuid.uuid4()),
        "incident_id": incident_id,
        "action": body.action,
        "reviewer": body.reviewer.strip(),
        "comment": body.comment.strip() if body.comment else None,
        "timestamp": now,
    }
    reviews.append(entry)
    REVIEWS_FILE.write_text(json.dumps(reviews, indent=2))
    new_status = _ACTION_TO_STATUS.get(entry["action"], entry["action"])
    return {"status": new_status, "recorded_at": now}
```

```javascript
// ── Submit review ──────────────────────────────────
async function submitReview(action) {
  const nameEl = document.getElementById('reviewer-name');
  const commentEl = document.getElementById('review-comment');
  const feedbackEl = document.getElementById('action-feedback');
  const name = nameEl.value.trim();
  const comment = commentEl.value.trim();

  nameEl.classList.remove('error');
  commentEl.classList.remove('error');
  feedbackEl.className = '';
  feedbackEl.textContent = '';

  if (!name) {
    nameEl.classList.add('error');
    feedbackEl.className = 'action-error';
    feedbackEl.textContent = 'Reviewer name is required.';
    return;
  }
  if (action === 'request_changes' && !comment) {
    commentEl.classList.add('error');
    feedbackEl.className = 'action-error';
    feedbackEl.textContent = 'Comment is required when requesting changes.';
    return;
  }

  const res = await fetch(`/postmortems/${_currentPMId}/review`, {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({action, reviewer: name, comment: comment || null}),
  });
  if (!res.ok) {
    feedbackEl.className = 'action-error';
    feedbackEl.textContent = 'Failed to save review. Try again.';
    return;
  }
  const data = await res.json();
  const label = action === 'approve' ? 'Approved' : action === 'request_changes' ? 'Changes requested' : 'Rejected';
  feedbackEl.className = 'action-success';
  feedbackEl.textContent = `✓ ${label} — recorded at ${new Date(data.recorded_at).toLocaleTimeString('en-US', {timeZone:'UTC'})} UTC`;
  // Refresh detail and list
  await openPMDetail(_currentPMId, null);
  await loadReviewList();
}
```
